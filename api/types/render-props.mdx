## RenderCellProps\<TRow, TSummaryRow\>

Props passed to custom cell renderers.

```typescript
interface RenderCellProps<TRow, TSummaryRow = unknown> {
  column: CalculatedColumn<TRow, TSummaryRow>;
  row: TRow;
  rowIdx: number;
  isCellEditable: boolean;
  tabIndex: number;
  onRowChange: (row: TRow) => void;
}
```

### Properties

<ParamField path="column" type="CalculatedColumn<TRow, TSummaryRow>" required>
  The column object for this cell.
</ParamField>

<ParamField path="row" type="TRow" required>
  The row object for this cell.
</ParamField>

<ParamField path="rowIdx" type="number" required>
  The row index.
</ParamField>

<ParamField path="isCellEditable" type="boolean" required>
  Whether the cell is editable.
</ParamField>

<ParamField path="tabIndex" type="number" required>
  The tab index for keyboard navigation.
</ParamField>

<ParamField path="onRowChange" type="(row: TRow) => void" required>

## RenderEditCellProps\<TRow, TSummaryRow\>

Props passed to custom edit cell renderers (editors).

```typescript
interface RenderEditCellProps<TRow, TSummaryRow = unknown> {
  column: CalculatedColumn<TRow, TSummaryRow>;
  row: TRow;
  rowIdx: number;
  onRowChange: (row: TRow, commitChanges?: boolean) => void;
  onClose: (commitChanges?: boolean, shouldFocusCell?: boolean) => void;
}
```

### Properties

<ParamField path="column" type="CalculatedColumn<TRow, TSummaryRow>" required>
  The column object for this cell.
</ParamField>

<ParamField path="row" type="TRow" required>
  The row object for this cell.
</ParamField>

<ParamField path="rowIdx" type="number" required>
  The row index.
</ParamField>

<ParamField path="onRowChange" type="(row: TRow, commitChanges?: boolean) => void" required>
  Callback to update the row. Pass `true` for `commitChanges` to close the editor.
</ParamField>

<ParamField path="onClose" type="(commitChanges?: boolean, shouldFocusCell?: boolean) => void" required>
  Callback to close the editor. Pass `true` for `commitChanges` to save changes, `false` to discard.
  Pass `true` for `shouldFocusCell` to return focus to the cell.
</ParamField>

```typescript
  selectedCellEditor: ReactElement<RenderEditCellProps<TRow>> | undefined;
  onRowChange: (column: CalculatedColumn<TRow, TSummaryRow>, rowIdx: number, newRow: TRow) => void;
  rowClass: Maybe<(row: TRow, rowIdx: number) => Maybe<string>>;
  isTreeGrid: boolean;
}
```

### Properties

<ParamField path="row" type="TRow" required>
  The row object.
</ParamField>

<ParamField path="viewportColumns" type="readonly CalculatedColumn<TRow, TSummaryRow>[]" required>
  Array of visible columns in the viewport.
</ParamField>

<ParamField path="rowIdx" type="number" required>
  The row index.
</ParamField>

<ParamField path="selectedCellIdx" type="number | undefined" required>
  Index of the selected cell in this row, or undefined if no cell is selected.
</ParamField>

<ParamField path="isRowSelected" type="boolean" required>
  Whether the row is selected.
</ParamField>

<ParamField path="isRowSelectionDisabled" type="boolean" required>
  Whether row selection is disabled for this row.
</ParamField>

<ParamField path="gridRowStart" type="number" required>

<ParamField path="lastFrozenColumnIndex" type="number" required>
  Index of the last frozen column.
</ParamField>

<ParamField path="draggedOverCellIdx" type="number | undefined" required>
  Index of the cell being dragged over, or undefined.
</ParamField>

<ParamField path="selectedCellEditor" type="ReactElement<RenderEditCellProps<TRow>> | undefined" required>
  The editor element for the selected cell, or undefined.
</ParamField>

<ParamField path="onRowChange" type="(column: CalculatedColumn<TRow, TSummaryRow>, rowIdx: number, newRow: TRow) => void" required>
  Callback to update the row.

<ParamField path="rowClass" type="(row: TRow, rowIdx: number) => Maybe<string>" required>

### Properties

  Whether the checkbox is disabled.

<ParamField path="tabIndex" type="number" required>
